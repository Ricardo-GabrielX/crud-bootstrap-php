/**
 * Criptografia
 * 
 * Função que criptografa a senha utilizando Blowfish
 */
function criptografia($senha) {
    /*
     * ==> Criptografia Blowfish
     * 
     * Referência: http://www.linhadecodigo.com.br/artigo/3532/criptografando-senhas-usando-bcrypt-blowfish-no-php.aspx
     */

    // Configurações de criptografia
    $custo = "08";
    $salt = "CflfllePArKlBJomM0F6aJ";

    // Gera um hash baseado em bcrypt
    $hash = crypt($senha, "$2a$" . $custo . "$" . $salt . "$");

    return $hash; // Retorna a senha criptografada
}

/**
 * Upload de imagens
 * 
 * Função para realizar o upload de arquivos de imagem
 */
function upload($pasta_destino, $arquivo_destino, $tipo_arquivo, $nome_temp, $tamanho_arquivo) {
    /*
     * --> Upload de arquivos no PHP
     * 
     * Referência: https://www.w3schools.com/php/php_file_upload.asp
     */

    try {
        // Nome do arquivo
        $nomearquivo = basename($arquivo_destino);

        // Inicializa o indicador de upload
        $uploadOk = 1;

        // Verificando se o arquivo é uma imagem
        $check = getimagesize($nome_temp);
        if ($check !== false) {
            $_SESSION['message'] = "O arquivo é uma imagem: " . $check["mime"] . ".";
            $_SESSION['type'] = "info";
            $uploadOk = 1;
        } else {
            $uploadOk = 0;
            throw new Exception("O arquivo não é uma imagem!");
        }

        // Verificando se o arquivo já existe na pasta
        if (file_exists($arquivo_destino)) {
            $uploadOk = 0;
            throw new Exception("Desculpe, o arquivo já existe!");
        }

        // Verificando se o tamanho do arquivo é maior que 5MB
        if ($tamanho_arquivo > 5000000) {
            $uploadOk = 0;
            throw new Exception("Desculpe, mas o arquivo é muito grande!");
        }

        // Permitindo apenas certos tipos de arquivo
        if ($tipo_arquivo != "jpg" && $tipo_arquivo != "png" && $tipo_arquivo != "jpeg" && $tipo_arquivo != "gif") {
            $uploadOk = 0;
            throw new Exception("Desculpe, mas só são permitidos arquivos de imagem JPG, JPEG, PNG e GIF!");
        }

        // Verifica se ocorreu algum erro com o upload
        if ($uploadOk == 0) {
            throw new Exception("Desculpe, mas o arquivo não pode ser enviado.");
        } else {
            // Se tudo estiver OK, tentamos fazer o upload do arquivo
            if (move_uploaded_file($nome_temp, $arquivo_destino)) {
                $_SESSION['message'] = "O arquivo " . htmlspecialchars($nomearquivo) . " foi armazenado com sucesso.";
                $_SESSION['type'] = "success";
            } else {
                throw new Exception("Desculpe, houve um erro ao enviar o arquivo.");
            }
        }
    } catch (Exception $e) {
        $_SESSION['message'] = "Aconteceu um erro: " . $e->getMessage();
        $_SESSION['type'] = "danger";
    }
}

// Função para cadastrar um novo usuário
function add() {
    if (!empty($_POST['usuario'])) {
        try {
            $usuario = $_POST['usuario'];

            // Verifica se foi enviada uma foto
            if (!empty($_FILES["foto"]["name"])) {
                // Pasta onde as fotos serão armazenadas
                $pasta_destino = "Fotos/";
                
                // Caminho completo do arquivo
                $arquivo_destino = $pasta_destino . basename($_FILES["foto"]["name"]);
                
                // Nome do arquivo
                $nomearquivo = basename($_FILES["foto"]["name"]);
                
                // Pegando informações da imagem
                $resolucao_arquivo = getimagesize($_FILES["foto"]["tmp_name"]);
                $tamanho_arquivo = $_FILES["foto"]["size"]; // Tamanho do arquivo em bytes
                $nome_temp = $_FILES["foto"]["tmp_name"]; // Nome e caminho temporário do arquivo no servidor
                $tipo_arquivo = strtolower(pathinfo($arquivo_destino, PATHINFO_EXTENSION)); // Extensão do arquivo

                // Chama a função de upload para gravar a foto
                upload($pasta_destino, $arquivo_destino, $tipo_arquivo, $nome_temp, $tamanho_arquivo);

                // Adiciona o nome do arquivo ao array do usuário
                $usuario['foto'] = $nomearquivo;
            }

            // Criptografando a senha
            if (!empty($usuario['password'])) {
                $senha = criptografia($usuario['password']);
                $usuario['password'] = $senha;
            }

            // Salva o usuário no banco de dados
            save("usuarios", $usuario);

            // Redireciona para a página de listagem de usuários
            header('Location: index.php');
        } catch (Exception $e) {
            $_SESSION['message'] = "Aconteceu um erro: " . $e->getMessage();
            $_SESSION['type'] = "danger";
        }
    }
}
